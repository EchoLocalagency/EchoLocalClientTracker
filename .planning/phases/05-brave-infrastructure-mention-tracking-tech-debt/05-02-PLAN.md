1. Create Supabase tables (mentions + competitor_aio_citations):
   ```sql
   CREATE TABLE IF NOT EXISTS mentions (
       id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
       client_id UUID NOT NULL REFERENCES clients(id) ON DELETE CASCADE,
       platform TEXT NOT NULL,
       source_url TEXT NOT NULL,
       source_domain TEXT NOT NULL,
       mention_type TEXT NOT NULL,
       title TEXT,
       snippet TEXT,
       discovered_at TIMESTAMPTZ DEFAULT NOW(),
       UNIQUE(client_id, source_url)
   );
   CREATE INDEX IF NOT EXISTS idx_mentions_client ON mentions(client_id);
   CREATE INDEX IF NOT EXISTS idx_mentions_platform ON mentions(platform);

   CREATE TABLE IF NOT EXISTS competitor_aio_citations (
       id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
       client_id UUID NOT NULL REFERENCES clients(id) ON DELETE CASCADE,
       keyword TEXT NOT NULL,
       competitor_url TEXT NOT NULL,
       competitor_domain TEXT NOT NULL,
       citation_title TEXT,
       first_seen TIMESTAMPTZ DEFAULT NOW(),
       last_seen TIMESTAMPTZ DEFAULT NOW(),
       UNIQUE(client_id, keyword, competitor_url)
   );
   CREATE INDEX IF NOT EXISTS idx_competitor_aio_client ON competitor_aio_citations(client_id);
   ```
